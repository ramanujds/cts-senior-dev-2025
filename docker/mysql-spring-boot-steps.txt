Create a docker network

docker network create app-network

Run MySQL container

docker run -d --name mysql-container \
--network=app-network \
-e MYSQL_ROOT_PASSWORD=password \
-e MYSQL_DATABASE=product_db \
-p 3306:3306 \
mysql:latest


Run Spring Boot application container

docker run -d --name product-app-container \
--network=app-network \
-e MYSQL_HOST=mysql-container \
-e MYSQL_PORT=3306 \
-e MYSQL_DB=product_db \
-e MYSQL_USER=root \
-e MYSQL_PASSWORD=password \
-p 5000:8080 \
product-app:latest
